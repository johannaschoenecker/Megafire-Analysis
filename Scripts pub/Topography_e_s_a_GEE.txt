// Load the NASADEM elevation dataset and select the 'elevation' band
var dem = ee.Image('NASA/NASADEM_HGT/001').select('elevation');

// Clip the DEM to the study area (outline)
var dem_SN = dem.clip(outline);
print("Clipped DEM:", dem_SN);

// ===================================================================
// Terrain Analysis: Slope & Aspect Calculation
// ===================================================================

// Compute the slope from the DEM (Units: Degrees, Range: [0,90])
var slope = ee.Terrain.slope(dem_SN);
print("Slope:", slope);

// Compute the aspect from the DEM (Units: Degrees, 0=N, 90=E, 180=S, 270=W)
var aspect = ee.Terrain.aspect(dem_SN);
print("Aspect:", aspect);

// ===================================================================
// Export Processed Datasets to Google Drive as GeoTIFFs
// ===================================================================

// Function to export images with predefined settings
function exportImage(image, description) {
  Export.image.toDrive({
    image: image.toFloat(),  // Convert to Float to ensure correct data format
    description: description, // Name of the output file
    scale: 30,  // Set spatial resolution to 30 meters
    region: outline,  // Define the export region (clipped study area)
    crs: 'EPSG:3310',  // California Albers projection (NAD83)
    maxPixels: 1e9,  // Allow large datasets to be exported
    fileFormat: 'GeoTIFF'  // Export as GeoTIFF format
  });
}

// Export DEM
exportImage(dem_SN, "elevation");

// Export Slope
exportImage(slope, "slope");

// Export Aspect
exportImage(aspect, "aspect");
